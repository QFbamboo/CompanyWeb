create table T_ProductCategories(
   Id INT NOT NULL AUTO_INCREMENT,
   Name VARCHAR(250) NOT NULL,
   PRIMARY KEY ( Id )
);

create table T_Product(
   Id INT NOT NULL AUTO_INCREMENT,
   Name VARCHAR(250) ,
   ImagePath varchar(250),
   Msg varchar(250), 
   CategoriyId int not null,
   PRIMARY KEY ( Id )
);
